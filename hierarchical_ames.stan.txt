// hierarchical_ames.stan
data {
  int<lower=1> N;                  // Number of observations
  int<lower=1> K;                  // Number of predictors (LASSO selected)
  matrix[N, K] X;                  // Design matrix (LASSO features)
  vector[N] y;                     // Outcome variable (log SalePrice)
  int<lower=1> G;                  // Number of neighborhoods
  int<lower=1, upper=G> gid[N];    // Neighborhood ID for each obs
}

parameters {
  vector[K] beta;                 // Coefficients for LASSO predictors
  real mu_alpha;                 // Hyper mean for neighborhood intercepts
  real<lower=0> sigma;           // Observation noise std dev
  real<lower=0> sigma_alpha;     // Variation std dev between neighborhoods
  vector[G] alpha_raw;            // Non-centered neighborhood intercepts (raw deviations)
}

transformed parameters {
  // Centered intercepts (derived from raw deviations + mean)
  vector[G] alpha = mu_alpha + alpha_raw * sigma_alpha;
}

model {
  // Priors (Non-centered parameterization for alpha)
  beta ~ normal(0, 5);          // Prior on LASSO feature coefficients (slightly stronger than 10 on scaled data)
  mu_alpha ~ normal(12, 2);     // Prior on overall mean log price (informed by EDA mean ~12)
  sigma ~ exponential(1);       // Prior on residual std dev (rate 1 implies mean 1)
  sigma_alpha ~ exponential(1); // Prior on std dev of neighborhood effects
  alpha_raw ~ std_normal();       // Prior on the raw deviations (standard normal)

  // Likelihood
  y ~ normal(alpha[gid] + X * beta, sigma);
}

generated quantities {
  vector[N] y_rep; // Replicated data for posterior predictive checks
  vector[N] log_lik; // Log likelihood for LOO/WAIC calculation
  for (n in 1:N) {
    real mu_n = alpha[gid[n]] + X[n] * beta;
    y_rep[n] = normal_rng(mu_n, sigma);
    log_lik[n] = normal_lpdf(y[n] | mu_n, sigma);
  }
}
